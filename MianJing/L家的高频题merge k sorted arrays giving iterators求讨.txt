public static class iteratorComp implements Comparator<Iterator<Integer>>{
        public int compare(Iterator<Integer> a, Iterator<Integer> b){
            int a1 = 0;
            int b1 = 0;
            if(a.hasNext()){
                a1 = a.next();
            }
            if(b.hasNext()){
                b1 = b.next();
            }
            return a1 - b1;
        }
    }
